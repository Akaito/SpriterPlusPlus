cmake_minimum_required(VERSION 2.8 FATAL_ERROR)

include_directories("${PROJECT_SOURCE_DIR}/SFML-2.3.2/include")

file(GLOB_RECURSE spp_SRC 
	${PROJECT_SOURCE_DIR}/example/*.h
	${PROJECT_SOURCE_DIR}/spriterengine/*.h
	${PROJECT_SOURCE_DIR}/tinyxml2/*.h
	${PROJECT_SOURCE_DIR}/pugixml/*.hpp
	${PROJECT_SOURCE_DIR}/example/*.cpp
	${PROJECT_SOURCE_DIR}/spriterengine/*.cpp
	${PROJECT_SOURCE_DIR}/tinyxml2/*.cpp
	${PROJECT_SOURCE_DIR}/pugixml/*.cpp)

link_directories("${PROJECT_SOURCE_DIR}/SFML-2.3.2/lib")

add_executable(SpriterPlusPlus ${spp_SRC})

set_property(TARGET SpriterPlusPlus PROPERTY PROJECT_LABEL "SpriterPlusPlus")

target_link_libraries (SpriterPlusPlus	
	optimized sfml-graphics.lib	debug sfml-graphics-d.lib
	optimized sfml-window.lib	debug sfml-window-d.lib
	optimized sfml-system.lib	debug sfml-system-d.lib
	optimized sfml-audio.lib	debug sfml-audio-d.lib)
